! Youtube - Hide any partially watched videos
! Now, hiding any partially watched videos on Home, Subscriptions, Channels and Watch part of YT.
! Uncomment Search and Playlists if you want to.


!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!
!!! PC WEB BROWSER EXTENSION !!!
!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!

! Home
! 2025-09 GUI
youtube.com##[page-subtype="home"] ytd-rich-item-renderer:has(yt-lockup-view-model:has(.ytThumbnailOverlayProgressBarHostWatchedProgressBarSegment))
! OLD GUI
youtube.com##[page-subtype="home"] ytd-thumbnail-overlay-resume-playback-renderer:upward(ytd-rich-item-renderer:has(#progress))

! Subscriptions
! 2025-09 GUI
youtube.com##[page-subtype="subscriptions"] ytd-rich-item-renderer:has(yt-lockup-view-model:has(.ytThumbnailOverlayProgressBarHostWatchedProgressBarSegment))
! OLD GUI
youtube.com##[page-subtype="subscriptions"] ytd-thumbnail-overlay-resume-playback-renderer:upward(ytd-rich-item-renderer:has(#progress))

! Channels - Videos tab
! 2025-09 GUI
youtube.com##[page-subtype="channels"] ytd-rich-item-renderer:has(yt-lockup-view-model:has(.ytThumbnailOverlayProgressBarHostWatchedProgressBarSegment))
! OLD GUI
youtube.com##[page-subtype="channels"] ytd-thumbnail-overlay-resume-playback-renderer:upward(ytd-rich-item-renderer:has(#progress))

! Channels - Home of a channel
! 2025-09 GUI
youtube.com##[page-subtype="channels"] ytd-grid-video-renderer:has(ytd-thumbnail-overlay-resume-playback-renderer:has(#progress))
! OLD GUI
youtube.com##[page-subtype="channels"] ytd-thumbnail-overlay-resume-playback-renderer:upward(ytd-grid-video-renderer:has(#progress))

! Watch
! 2025-09 GUI
[$path=/watch]youtube.com## yt-lockup-view-model:has(.ytThumbnailOverlayProgressBarHostWatchedProgressBarSegment)
! OLD GUI
[$path=/watch]youtube.com## ytd-thumbnail-overlay-resume-playback-renderer:upward(ytd-compact-video-renderer:has(#progress))

! Search
![$path=/results]youtube.com## ytd-thumbnail-overlay-resume-playback-renderer:upward(ytd-video-renderer:has(#progress))

! Playlists but not 'Watch Later'
!youtube.com##[page-subtype="playlist"]:not(:matches-path(WL)) ytd-thumbnail-overlay-resume-playback-renderer:upward(ytd-playlist-video-renderer:has(#progress))
